Chapter 3.1: Basic Rendering

Paste this code in console. 

/* execute this first */
var app = require('app');

var movie = app.movies.get(1);
var view = new app.MovieView({model: movie});
document.body.appendChild(view.render().el);

/* execute to select movie */
app.movies.selectByID(1);
view.render().el;

/* execute to unselect movie */
app.movies.resetSelected();
view.render().el;
